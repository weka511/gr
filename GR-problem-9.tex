\begin{theorem}
	\begin{align*}
	0 =& D_{\mu} T^{\mu\nu} \\
	\equiv & \partial_{\mu} T^{\mu\nu} + \Gamma^{\mu}_{\beta\mu}T^{\beta}_{\nu}-\Gamma^{\beta}_{\nu\mu}T^{\mu}_{\beta}\\
	\equiv& \frac{1}{\sqrt(|g|)}\partial_{\mu}(T^{\mu}_{\nu}\sqrt{(|g|)}) - \frac{1}{2}(\partial_{\mu}g_{\mu\alpha})T^{\mu\alpha}
	\end{align*}
